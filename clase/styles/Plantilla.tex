\newcommand*\chapterlabel{}
\titleformat{\chapter}
  {\gdef\chapterlabel{}
   \normalfont\sffamily\Huge\bfseries\scshape}
  {\gdef\chapterlabel{\thechapter\ }}{0pt}
  {\begin{tikzpicture}[remember picture,overlay]
      \node[yshift=-3cm,xshift=-3cm] at (current page.north west)
      {\begin{tikzpicture}[remember picture, overlay]
        \draw[blue!8!white,fill=yourcolor] (2,0) rectangle
          (1.3\paperwidth,3cm);
        \node[anchor=east,xshift=\paperwidth,rectangle,
              rounded corners=20pt,inner sep=11pt,
              fill=blue!50!black]
              {\color{white}{\large}\chapterlabel\hspace{-10pt}. #1};
       \end{tikzpicture}
      };
   \end{tikzpicture}
  }
  %%%%%%%%%%%%%%%%%%%%%%%%%%%%%% capítulo sin enumerar %%%%%%%%%%%%%%%%%%%%%%%%
  \titleformat{name=\chapter,numberless}
